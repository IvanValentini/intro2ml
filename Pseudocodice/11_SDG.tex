	\Require : Initial Parameter $\theta$\;

		\comment{Compute gradient estimate over sample example ($x^{(i)}, y^{(i)}$) from training set}
		g = $+\nabla_\theta\sum_i L(f(x^{(i)};\theta),y^{(i)}$\;
		\comment{Apply update}
		$\theta = \theta -\varepsilon_k g$\;
	}
