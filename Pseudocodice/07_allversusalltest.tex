\For{$i\ =\ 1$ \To $K\ -\ 1$}{
	\For{$j\ =\ i\ +\ 1$ \To $K$}{
		y = $f_{ij}(\hat{x})$\;
		$score_i\ =\ score_i + y$\;
		$score_j\ =\ score_j - y$\;
	}
}
